38static int FileGeneration_checkFileExisted(FileGeneration* oFG, unsigned char* filePath)
39{
40 // Variables for obtaining the project path and file name location(index)
41 unsigned char projectPath[FILE_GENERATION_PROJECT_PATH_SIZE];
42 unsigned char* fileNameLoc = NULL;
43 FileGeneration_getProjectPath(oFG, projectPath);
44
45 // Parsing the filePath, and generate the folder location and file name
46 // Finding the last appearance of the character '/'
47 char* loc = NULL;
48 loc = strrchr((char*)filePath, '/');
49 int httpStatus = 200;
50
51 // If strrchr finds nothing, then the address of fileNameLoc is equal to the one of filePath.
52 if (loc == NULL) {
53 fileNameLoc = filePath;
54 } else { // Concatenate substring of filePath to projectPath
55 int length = 0;
56 length = (int)strlen((char*)projectPath);
57 if(length < FILE_GENERATION_PROJECT_PATH_SIZE) {
58 projectPath[length++] = (unsigned char)'/';
59 } else {
60 httpStatus = 500;
61 return httpStatus;
62 }
63
64 for (int i = 0; i < (((unsigned char*)loc) - filePath); i++) {
65 // (((unsigned char*)loc) - filePath) implies the address
66 projectPath[length] = *(filePath + i);
67 length++;
68 }
69 projectPath[length] = (unsigned char)'\0';
70 fileNameLoc = (unsigned char*)(loc + 1);
71 }
72 // Now, the projectPath will contain the folders' names of the filePath and
73 // the last two characters are (1) any character except '/' and (2) '\0'.
74 // e.g., the projectPath is "/C/home/user/myproject\0"
75
76 httpStatus = 500;
77#if defined(_WIN32) || defined(_WIN64)
78 // Windows
79 int length = 0;
80 length = (int)strlen((char*)projectPath);
81 projectPath[length++] = '/';
82 projectPath[length++] = '*';
83 projectPath[length++] = '\0';
84
85 WIN32_FIND_DATA data;
86 HANDLE handler = NULL;
87
88 // If there exist files in the directory
89 if ((handler = FindFirstFile((char*)projectPath, &data)) != INVALID_HANDLE_VALUE) {
90 do {
91 char* individualFile = data.cFileName;
92 if (strcmp(individualFile, ".") == 0 || strcmp(individualFile, "..") == 0) {
93 continue;
94 }
95 // Assemble the file name
96 unsigned char tempProjectPath[FILE_GENERATION_PROJECT_PATH_SIZE];
97 memcpy(tempProjectPath, projectPath, (sizeof(unsigned char) * sizeof(projectPath)));
98 int length = 0;
99 length = (int)strlen((char*)tempProjectPath);
100 tempProjectPath[length - 1] = (unsigned char)'\0';
101 strcat((char*)tempProjectPath, (char*)fileNameLoc);
102
103 // Determining items; there are two types of items; one is folder type, the other is file type.
104 const int dir = (data.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) != 0;
105 if (dir == 1) {
106 continue;
107 } else {
108 if (strcmp(individualFile, (char*)fileNameLoc) == 0) {
109 httpStatus = 200;
110 break;
111 }
112 }
113
114 } while (FindNextFile(handler, &data) != 0);
115 FindClose(handler);
116 }
117#else
118 // Linux
119 DIR* dirCurrent = opendir((char*)projectPath);
120 if (dirCurrent == NULL) {
121 httpStatus = 500;
122 return httpStatus;
123 }
124 const struct dirent* dirent = NULL;
125 struct stat status;
126
127 // Iterations of the directory that users specified
128 while ((dirent = readdir(dirCurrent)) != NULL) {
129 // Fetching all the files in the directory
130 const char* individualFile = dirent->d_name;
131 if (strcmp(individualFile, ".") == 0 || strcmp(individualFile, "..") == 0) {
132 continue;
133 }
134
135 // Simulations of the path of the directory
136 unsigned char tempProjectPath[FILE_GENERATION_PROJECT_PATH_SIZE];
137 memcpy(tempProjectPath, projectPath, (sizeof(unsigned char) * sizeof(projectPath)));
138 int length = 0;
139 length = (int)strlen((char*)tempProjectPath);
140 tempProjectPath[length++] = (unsigned char)'/';
141 tempProjectPath[length] = (unsigned char)'\0';
142 strcat((char*)tempProjectPath, (char*)fileNameLoc);
143
144 // If the file path does not exist (stat will return -1; otherwise 0).
145 if (stat((char*)tempProjectPath, &status) == -1) {
146 continue;
147 }
148
149 // Determining the stuff of the path shall not belong to the folder type
150 const int isDir = (status.st_mode & __S_IFDIR) != 0;
151 if (isDir == 0) {
152 httpStatus = 200;
153 break;
154 }
155 }
156 closedir(dirCurrent);
157#endif
158 return httpStatus;
159}
