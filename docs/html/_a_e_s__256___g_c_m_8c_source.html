55static int AES_256_GCM_encryption(
56 Encryption* pEnc,
57 const unsigned char* plaintext,
58 const int plaintextLen,
59 unsigned char* ciphertext,
60 int* ciphertextLen)
61{
62 int httpStatus = 500;
63 AES_256_GCM* pA2gObject = (AES_256_GCM*)pEnc;
64 unsigned char const* key = pA2gObject->masterKey;
65 unsigned char authTag[AES_256_GCM_TAG_SIZE];// The tag for the ciphertext which can verify if the ciphertext had been modified in connection.
66
67 httpStatus = AES_256_GCM_checkFileExisted(pA2gObject);
68 if (httpStatus != 200) {
69 //fprintf(stderr, "Key Lost.\n");
70 return 500;
71 }
72
73 unsigned char iv[(AES_256_GCM_IV_SIZE + 1)]; // "+1" is for the symbol '\0'.
74 memset(iv, (unsigned char)'\0', AES_256_GCM_IV_SIZE + 1);
75 httpStatus = AES_256_GCM_getIV(iv);
76
77 if (httpStatus == 200) {
78
79 int currentLen = 0;
80 int cipherLen = 0;
81
82 EVP_CIPHER_CTX* ecCtx;
83
84 // New an instance
85 if (!(ecCtx = EVP_CIPHER_CTX_new())) {
86 //fprintf(stderr, "EVP_CIPHER_CTX_new\n");
87 return 500;
88 }
89
90 // Initialize encryption approach
91 if (EVP_EncryptInit_ex(ecCtx, EVP_aes_256_gcm(), NULL, NULL, NULL) != 1) {
92 //fprintf(stderr, "EVP_EncryptInit_ex\n");
93 EVP_CIPHER_CTX_free(ecCtx);
94 return 500;
95 }
96
97 // Set the GCM with the IV_SIZE
98 if (EVP_CIPHER_CTX_ctrl(ecCtx, EVP_CTRL_GCM_SET_IVLEN, AES_256_GCM_IV_SIZE, NULL) != 1) {
99 //fprintf(stderr, "EVP_CIPHER_CTX_ctrl\n");
100 EVP_CIPHER_CTX_free(ecCtx);
101 return 500;
102 }
103
104 // Set the encryption key and iv values
105 if (EVP_EncryptInit_ex(ecCtx, NULL, NULL, key, iv) != 1) {
106 //fprintf(stderr, "EVP_EncryptInit_ex\n");
107 EVP_CIPHER_CTX_free(ecCtx);
108 return 500;
109 }
110
111 // Encryption process
112 if (EVP_EncryptUpdate(ecCtx, ciphertext, &currentLen, plaintext, plaintextLen) != 1) {
113 //fprintf(stderr, "EVP_EncryptUpdate\n");
114 EVP_CIPHER_CTX_free(ecCtx);
115 return 500;
116 }
117 cipherLen = currentLen;
118
119 // The last encryption of GCM
120 if (EVP_EncryptFinal_ex(ecCtx, ciphertext + currentLen, &currentLen) != 1) {
121 //fprintf(stderr, "EVP_EncryptFinal_ex\n");
122 EVP_CIPHER_CTX_free(ecCtx);
123 return 500;
124 }
125 cipherLen += currentLen;
126
127 // Insert authentication tag
128 if (EVP_CIPHER_CTX_ctrl(ecCtx, EVP_CTRL_GCM_GET_TAG, 16, authTag) != 1) {
129 //fprintf(stderr, "EVP_CIPHER_CTX_ctrl\n");
130 EVP_CIPHER_CTX_free(ecCtx);
131 return 500;
132 }
133
134 // Free the instance
135 EVP_CIPHER_CTX_free(ecCtx);
136 *ciphertextLen = cipherLen;
137
138 // Adding the IV and auth values into variable sequentially, cipherText and modifying the variable, cipherTextLen
139 // Due to SonarLint's suggestions, the size of the array shall be larger than 1.
140 unsigned char tmpCipher[((unsigned int)cipherLen) + 1];
141 memset(tmpCipher, (unsigned char)'\0', ((unsigned int)cipherLen) + 1);
142 memcpy(tmpCipher, ciphertext, cipherLen);
143 memset(ciphertext, (unsigned char)'\0', cipherLen);
144 memcpy(ciphertext, iv, AES_256_GCM_IV_SIZE);
145 memcpy((ciphertext + AES_256_GCM_IV_SIZE), authTag, AES_256_GCM_TAG_SIZE);
146 memcpy((ciphertext + AES_256_GCM_IV_SIZE + AES_256_GCM_TAG_SIZE), tmpCipher, cipherLen);
147 *ciphertextLen = *ciphertextLen + AES_256_GCM_IV_SIZE + AES_256_GCM_TAG_SIZE;
148
149 return 200;
150 } else {
151 return 500;
152 }
153}

166static int AES_256_GCM_decryption(
167 Encryption* pEnc,
168 unsigned char* ciphertext,
169 int ciphertextLen,
170 unsigned char* plaintext,
171 int* plaintextLen
172 )
173{
174
175 AES_256_GCM* pA2gObject = (AES_256_GCM*)pEnc;
176 unsigned char const* key = pA2gObject->masterKey;
177 unsigned char authTag[AES_256_GCM_TAG_SIZE];// The tag for the ciphertext which can verify if the ciphertext had been modified in connection.
178
179 int httpStatus = 500;
180 httpStatus = AES_256_GCM_checkFileExisted(pA2gObject);
181 if (httpStatus != 200) {
182 //fprintf(stderr, "Key Lost.\n");
183 return 500;
184 }
185
186 // Obtaining the IV value
187 unsigned char iv[AES_256_GCM_IV_SIZE + 1];
188 memset(iv, '\0', AES_256_GCM_IV_SIZE + 1);
189 memcpy(iv, ciphertext, AES_256_GCM_IV_SIZE);
190
191 // Obtaining the authTag value
192 memset(authTag, '\0', AES_256_GCM_TAG_SIZE);
193 memcpy(authTag, ciphertext + AES_256_GCM_IV_SIZE, AES_256_GCM_TAG_SIZE);
194
195 // Removing the IV value from the variable, "ciphertext"
196 unsigned char tmpCipher[ciphertextLen + 1];
197 memset(tmpCipher, '\0', ciphertextLen + 1);
198
199 // Copying and ignoring the IV value
200 memcpy(tmpCipher, (ciphertext + AES_256_GCM_IV_SIZE + AES_256_GCM_TAG_SIZE), (ciphertextLen - AES_256_GCM_IV_SIZE - AES_256_GCM_TAG_SIZE));
201 memset(ciphertext, '\0', ciphertextLen);
202 memcpy(ciphertext, tmpCipher, (ciphertextLen - AES_256_GCM_IV_SIZE - AES_256_GCM_TAG_SIZE));
203 ciphertextLen = (ciphertextLen - AES_256_GCM_IV_SIZE - AES_256_GCM_TAG_SIZE);
204
205 int currentLen = 0;
206 int plainLen = 0;
207 int decryptedStatus = 0;
208
209 EVP_CIPHER_CTX* ecCtx;
210
211 // Creating an instance
212 if (!(ecCtx = EVP_CIPHER_CTX_new())) {
213 //fprintf(stderr, "EVP_CIPHER_CTX_new\n");
214 return 500;
215 }
216
217 // Initializing the decryption operation
218 if (!EVP_DecryptInit_ex(ecCtx, EVP_aes_256_gcm(), NULL, NULL, NULL)) {
219 //fprintf(stderr, "EVP_DecryptInit_ex\n");
220 EVP_CIPHER_CTX_free(ecCtx);
221 return 500;
222 }
223
224 // Setting iv length
225 if (!EVP_CIPHER_CTX_ctrl(ecCtx, EVP_CTRL_GCM_SET_IVLEN, AES_256_GCM_IV_SIZE, NULL)) {
226 //fprintf(stderr, "EVP_CIPHER_CTX_ctrl\n");
227 EVP_CIPHER_CTX_free(ecCtx);
228 return 500;
229 }
230
231 // Initializing key and iv
232 if (!EVP_DecryptInit_ex(ecCtx, NULL, NULL, key, iv)) {
233 //fprintf(stderr, "EVP_DecryptInit_ex\n");
234 EVP_CIPHER_CTX_free(ecCtx);
235 return 500;
236 }
237
238 // Providing the authenticated tag
239 if (!EVP_CIPHER_CTX_ctrl(ecCtx, EVP_CTRL_GCM_SET_TAG, AES_256_GCM_TAG_SIZE, authTag)) {
240 //fprintf(stderr, "EVP_CIPHER_CTX_ctrl\n");
241 EVP_CIPHER_CTX_free(ecCtx);
242 return 500;
243 }
244
245 // Decrypting plaintext
246 if (!EVP_DecryptUpdate(ecCtx, plaintext, &currentLen, ciphertext, ciphertextLen)) {
247 //fprintf(stderr, "EVP_DecryptUpdate\n");
248 EVP_CIPHER_CTX_free(ecCtx);
249 return 500;
250 }
251 plainLen = currentLen;
252
253 // The last decryption phase of GCM
254 decryptedStatus = EVP_DecryptFinal_ex(ecCtx, plaintext + currentLen, &currentLen);
255 plainLen += currentLen;
256
257 // free the instance
258 EVP_CIPHER_CTX_free(ecCtx);
259
260 *plaintextLen = plainLen;
261
262 // Checking if the decrypted result was successful
263 if (decryptedStatus > 0) {
264 return 200;
265 } else {
266 //fprintf(stderr, "decrypted Status error\n");
267 return 500;
268 }
269}
