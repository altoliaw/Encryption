34static int UU_Encode_encoder(const Encoder* oEncoder,
35 const unsigned char* plainText, int plainLen, unsigned char* encodedText, int* encodedTextLen)
36{
37 /*
38 * Reserving a character size (8 bit) in advance,
39 * and the buffer is modeled as a cyclic storage;
40 * as a result, the size of the inputBitBuffer equals to 2 * DECIMAL_BIT_STEP
41 */
42 unsigned char inputBitBuffer[(DECIMAL_BIT_STEP + DECIMAL_BIT_STEP)];
43 int totalBits = (plainLen * DECIMAL_BIT_STEP); // The size of the total bits of plainText
44 // Prediction of the encoded length
45 *encodedTextLen = (totalBits % UU_BIT_STEP == 0) ? (totalBits / UU_BIT_STEP) : (totalBits / UU_BIT_STEP + 1);
46 encodedText[*encodedTextLen + 1] = (unsigned char)'\0'; // The size is equal to `*encodedTextLen + 1` because the last element is for the '\0'.
47
48 int startElementIndex = 0; // The index of the starting element
49 int endElementIndex = 0; // The index of the last element
50 memset(inputBitBuffer, '\0', 2 * DECIMAL_BIT_STEP);
51 int remainder = 0; // The remainder elements in the stack
52 int transIndex = 0; // The index for inputBitBuffer
53
54 // Loop over all plainText characters
55 for (int i = 0; i < plainLen; i++) {
56 // Obtaining the ASCII representation
57 int decimal = (int)plainText[i];
58 // Prediction the starting index to the end index
59 endElementIndex = (startElementIndex + remainder + DECIMAL_BIT_STEP - 1) % (2 * DECIMAL_BIT_STEP);
60 // Putting a ASCII character into the bit storage;
61 // the approach is used by the right-left putting approach
62 for (int j = 0; j < DECIMAL_BIT_STEP; j++) {
63 int index = (endElementIndex - j) % (2 * DECIMAL_BIT_STEP);
64 int rightMostBit = decimal & 1;
65 inputBitBuffer[index] = (unsigned char)('0' + rightMostBit); // The bit
66 decimal = decimal >> 1; // Right shift a bit
67 }
68 remainder += DECIMAL_BIT_STEP; // The remainder is equal to the elements which are pushed.
69
70 // Fetch and become the char if the remainder is larger than or equal to UU_BIT_STEP
71 for (; remainder >= UU_BIT_STEP;
72 startElementIndex = (startElementIndex + UU_BIT_STEP) % (2 * DECIMAL_BIT_STEP), remainder -= UU_BIT_STEP) {
73
74 // 6 bits -> int
75 int sixBitsResult = 0;
76 for (int j = 0, k = startElementIndex; j < UU_BIT_STEP; j++, k++) {
77 k = k % (2 * DECIMAL_BIT_STEP);
78 sixBitsResult += ((int)(inputBitBuffer[k] - '0') << (UU_BIT_STEP - j - 1));
79 }
80 // If the sixBitsResult is equal to 0, the character is char(96), "`".
81 encodedText[transIndex++] = (sixBitsResult == 0) ? (unsigned char)(char)96 : (unsigned char)(char)(sixBitsResult + 32);
82 }
83 }
84
85 // If the remainder is not equal to 0, the last element shall be encoded with 0-bit(s).
86 if (remainder != 0) {
87 int lastElement = 0; // The value of the last element
88 for (int j = 0, k = startElementIndex; j < remainder; j++, k++) {
89 k = k % (2 * DECIMAL_BIT_STEP);
90 lastElement += ((inputBitBuffer[k] - '0') << (UU_BIT_STEP - j - 1));
91 }
92 // If the sixBitsResult is equal to 0, the character is char(96), "`".
93 encodedText[transIndex++] = (lastElement == 0) ? (unsigned char)(char)96 : (unsigned char)(char)(lastElement + 32);
94 }
95 encodedText[transIndex] = '\0';
96 return 200;
97}

109static int UU_Encode_decoder(const Encoder* oEncoder,
110 const unsigned char* encodedText, int encodedLen, unsigned char* plainText, int* plainTextLen)
111{
112 const unsigned char* encodedStart = encodedText;
113 /*
114 * Reserving a character size (6 bit) in advance,
115 * and the buffer is modeled as a cyclic storage
116 */
117 unsigned char inputBitBuffer[(UU_BIT_STEP + UU_BIT_STEP)];
118 int totalBits = (encodedLen * UU_BIT_STEP);
119
120 *plainTextLen = totalBits / DECIMAL_BIT_STEP;
121
122 // The length of the array is equal to *encodedLen + 1 (the last element is for the '\0'.)
123 plainText[*plainTextLen + 1] = (unsigned char)'\0';
124
125 int startElementIndex = 0; // The index of the starting element
126 int endElementIndex = 0; // The index of the last element
127 memset(inputBitBuffer, '\0', 2 * UU_BIT_STEP);
128 int remainder = 0; // The remainder elements in the stack
129 int transIndex = 0; // The index for inputBitBuffer
130
131 for (int i = 0; i < encodedLen; i++) {
132 // If the element is equal to the char(96), "`" in six-encoded result, the value is equal to the 0.
133 int decimal = ((int)encodedStart[i] == 96) ? 0 : (int)encodedStart[i] - 32;
134 // Prediction of the start index to the end index
135 endElementIndex = (startElementIndex + remainder + UU_BIT_STEP - 1) % (2 * UU_BIT_STEP);
136
137 // Putting a character into the bit storage
138 for (int j = 0; j < UU_BIT_STEP; j++) {
139 int index = (endElementIndex - j) % (2 * UU_BIT_STEP);
140 int rightMostBit = decimal & 1;
141 inputBitBuffer[index] = (unsigned char)('0' + rightMostBit);
142 decimal = decimal >> 1; // Shifting a right bit
143 }
144 remainder += UU_BIT_STEP;
145
146 // For handling bits to eight-encoded process
147 for (; remainder >= DECIMAL_BIT_STEP;
148 startElementIndex = (startElementIndex + DECIMAL_BIT_STEP) % (2 * UU_BIT_STEP), remainder -= DECIMAL_BIT_STEP) {
149 // 6 bits -> int
150 int eightBitsResult = 0;
151 for (int j = 0, k = startElementIndex; j < DECIMAL_BIT_STEP; j++, k++) {
152 k = k % (2 * UU_BIT_STEP);
153 eightBitsResult += ((inputBitBuffer[k] - '0') << (DECIMAL_BIT_STEP - j - 1));
154 }
155 plainText[transIndex++] = (char)eightBitsResult;
156 }
157 }
158
159 plainText[transIndex] = '\0';
160 return 200;
